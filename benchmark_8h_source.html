202 template<typename InputType, typename Function>
203 inline long double runtime(
204 Function func, const std::vector<InputType>& input) {
205
206 if (input.size() == 0)
207 return 0.0;
208
209 // Dummy variable
210 __volatile__ auto c = func(input[0]);
211
212 timer t = timer();
213
214 for (unsigned int j = 0; j < input.size(); ++j)
215 c += func(input[j]);
216
217 return t();
218 }

230 template<typename InputType = double, typename Function>
231 inline void benchmark(
232 const std::string& name,
233 Function func,
234 const std::vector<InputType>& input,
235 unsigned int runs = 0,
236 bool quiet = false) {
237
238 if (runs == 0)
239 runs = settings.defaultRuns;
240
241 // Whether the benchmark failed because of an exception
242 bool failed = false;
243
244 // Running average
245 long double averageRuntime;
246
247 // Running total sum of squares
248 long double sumSquares;
249
250 // Total runtime
251 long double totalRuntime;
252
253 try {
254
255 // Use Welford's algorithm to compute the average and the variance
256 totalRuntime = runtime(func, input);
257 averageRuntime = totalRuntime / input.size();
258 sumSquares = 0.0;
259
260 for (unsigned int i = 1; i < runs; ++i) {
261
262 // Compute the runtime for a single run
263 // and update the running estimates
264 const long double currentRun = runtime(func, input);
265 const long double currentAverage = currentRun / input.size();
266 totalRuntime += currentRun;
267
268 const long double tmp = averageRuntime;
269 averageRuntime = tmp + (currentAverage - tmp) / (i + 1);
270 sumSquares += (currentAverage - tmp)
271 * (currentAverage - averageRuntime);
272 }
273
274 } catch(...) {
275
276 // Catch any exception and mark the benchmark as failed
277 failed = true;
278 }
279
280 benchmark_result res {};
281 res.name = name;
282 res.runs = runs;
283 res.iterations = input.size();
284 res.totalRuntime = totalRuntime;
285 res.averageRuntime = averageRuntime;
286 res.runsPerSecond = 1000.0 / res.averageRuntime;
287 res.failed = failed;
288 res.quiet = quiet;
289
290 if (runs > 1)
291 res.stdevRuntime = std::sqrt(sumSquares / (runs - 1));
292
293 results.totalBenchmarks++;
294 if(failed)
295 results.failedBenchmarks++;
296
297 results.benchmarkResults[name].push_back(res);
298 }

307 template <typename InputType = double, typename Function>
308 inline void benchmark(
309 const std::string& name,
310 Function func,
311 const benchmark_options<InputType>& opt) {
312
313
314 // Generate input set
315 random::random_source rnd = random->get_rnd();
316 std::vector<InputType> input (opt.iterations);
317
318 for (unsigned int i = 0; i < opt.iterations; ++i)
319 input[i] = opt.inputGenerator(rnd);
320
321 // Benchmark over input set
322 benchmark(name, func, input, opt.runs, opt.quiet);
323 }
