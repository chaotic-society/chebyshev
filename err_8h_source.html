122 inline void setup(
123 const std::string& moduleName,
124 int argc = 0, const char** argv = nullptr) {
125
126 // Initialize other modules
127 settings = err_settings();
128 output = std::make_shared<output::output_context>();
129 random = std::make_shared<random::random_context>();
130
131 if(argc && argv)

133 settings.pickedChecks[argv[i]] = true;

135 std::cout << "Starting error checking on ";

147 inline void terminate(bool exit = false) {
148
149 unsigned int failedChecks = 0;
150 unsigned int totalChecks = 0;
151
152 for (const auto& pair : assertResults) {
153 for (const auto& testCase : pair.second) {
154 totalChecks++;
155 failedChecks += testCase.failed ? 1 : 0;
156 }
157 }
158
159 for (const auto& pair : errnoResults) {
160 for (const auto& testCase : pair.second) {
161 totalChecks++;
162 failedChecks += testCase.failed ? 1 : 0;
163 }
164 }
165
166 for (const auto& pair : exceptionResults) {
167 for (const auto& testCase : pair.second) {
168 totalChecks++;
169 failedChecks += testCase.failed ? 1 : 0;
170 }
171 }
172
173 // Ensure that an output file is specified
174 if( settings.outputToFile &&
175 !output->settings.outputFiles.size() &&
176 !settings.assertOutputFiles.size() &&
177 !settings.errnoOutputFiles.size() &&
178 !settings.exceptionOutputFiles.size() &&
179 !settings.outputFiles.size()) {
180
181 settings.outputFiles = { settings.moduleName + "_results" };
182 }
183
184 std::vector<std::string> outputFiles;
185
186 // Print assert results
187 outputFiles = settings.outputFiles;
188 outputFiles.insert(
189 outputFiles.end(),
190 settings.assertOutputFiles.begin(),
191 settings.assertOutputFiles.end()
192 );
193
194
195 output->print_results(
196 assertResults,
197 settings.assertColumns,
198 outputFiles
199 );
200
201 // Print errno checking results
202 outputFiles = settings.outputFiles;
203 outputFiles.insert(
204 outputFiles.end(),
205 settings.errnoOutputFiles.begin(),
206 settings.errnoOutputFiles.end()
207 );
208
209
210 output->print_results(
211 errnoResults,
212 settings.errnoColumns,
213 outputFiles
214 );
215
216 // Print exception checking results
217 outputFiles = settings.outputFiles;
218 outputFiles.insert(
219 outputFiles.end(),
220 settings.exceptionOutputFiles.begin(),
221 settings.exceptionOutputFiles.end()
222 );
223
224
225 output->print_results(
226 exceptionResults,
227 settings.exceptionColumns,
228 outputFiles
229 );
230
231 // Print overall checks results
232 std::cout << "Finished testing " << settings.moduleName << '\n';
233 std::cout << totalChecks << " total checks, ";
234 std::cout << failedChecks << " failed";
235
236 // Print proportion of failed checks, avoiding division by zero
237 if (totalChecks > 0) {
238
239 const double percent = (failedChecks / (double) totalChecks) * 100;
240 std::cout << " (" << std::setprecision(3) << percent << "%)" << std::endl;
241
242 } else {
243 std::cout << "No checks were run!" << std::endl;
244 }
245
246 if(exit) {
247 output->terminate();
248 std::exit(failedChecks);
249 }
250
251 wasTerminated = true;
252 }

301 template<typename Function, typename InputType>
302 inline void errno_value(
303 const std::string& name,
304 Function f,
305 InputType x,
306 int expected_errno,
307 bool quiet = false) {
308
309 errno_result res {};
310 errno = 0;
311
312 try {
313 volatile auto r = f(x);
314 r = *(&r);
315 } catch(...) {}
316
317 res.name = name;
318 res.evaluated = errno;
319 res.expectedFlags = { expected_errno };
320 res.failed = (errno != expected_errno);
321 res.quiet = quiet;
322
323 errnoResults[name].push_back(res);
324 }

368
369 res.name = name;
370 res.evaluated = errno;
371 res.expectedFlags = expected_flags;
372 res.quiet = quiet;

374 res.failed = false;
