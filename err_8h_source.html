130 inline void setup(

155 inline void terminate(bool exit = true) {
156
157 // Ensure that an output file is specified
158 if( settings.outputToFile &&
159 !output->settings.outputFiles.size() &&
160 !settings.assertOutputFiles.size() &&
161 !settings.errnoOutputFiles.size() &&
162 !settings.exceptionOutputFiles.size() &&
163 !settings.outputFiles.size()) {
164
165 settings.outputFiles = { settings.moduleName + "_results" };
166 }
167
168 std::vector<std::string> outputFiles;
169
170 // Print assert results
171 outputFiles = settings.outputFiles;
172 outputFiles.insert(
173 outputFiles.end(),
174 settings.assertOutputFiles.begin(),
175 settings.assertOutputFiles.end()
176 );
177
178
179 output->print_results(
180 results.assertResults,
181 settings.assertColumns,
182 outputFiles
183 );
184
185 // Print errno checking results
186 outputFiles = settings.outputFiles;
187 outputFiles.insert(
188 outputFiles.end(),
189 settings.errnoOutputFiles.begin(),
190 settings.errnoOutputFiles.end()
191 );
192
193
194 output->print_results(
195 results.errnoResults,
196 settings.errnoColumns,
197 outputFiles
198 );
199
200 // Print exception checking results
201 outputFiles = settings.outputFiles;
202 outputFiles.insert(
203 outputFiles.end(),
204 settings.exceptionOutputFiles.begin(),
205 settings.exceptionOutputFiles.end()
206 );
207
208
209 output->print_results(
210 results.exceptionResults,
211 settings.exceptionColumns,
212 outputFiles
213 );
214
215 // Print overall test results
216 std::cout << "Finished testing " << settings.moduleName << '\n';
217 std::cout << results.totalChecks << " total tests, ";
218 std::cout << results.failedChecks << " failed";
219
220 // Print proportion of failed test, avoiding division by zero
221 if (results.totalChecks > 0) {
222 std::cout << " (" << std::setprecision(3);
223 std::cout << (results.failedChecks / (double) results.totalChecks) * 100;
224 std::cout << "%)";
225 }
226 std::cout << std::endl;
227
228 // Discard previous results
229 results = err_results();
230
231 if(exit) {
232 output->terminate();
233 std::exit(results.failedChecks);
234 }
235 }

287 template<typename Function, typename InputType>
288 inline void check_errno(
289 const std::string& name,
290 Function f,
291 InputType x,
292 int expected_errno,
293 bool quiet = false) {
294
295 errno_result res {};
296 errno = 0;
297
298 try {
299 volatile auto r = f(x);
300 r = *(&r);
301 } catch(...) {}
302
303 res.name = name;
304 res.evaluated = errno;
305 res.expectedFlags = { expected_errno };
306 res.failed = (errno != expected_errno);
307 res.quiet = quiet;
308
309 results.totalChecks++;
310
311 if(res.failed)
312 results.failedChecks++;
313
314 results.errnoResults[name].push_back(res);
315 }

345 Function f,

358
359 res.name = name;
360 res.evaluated = errno;
361 res.expectedFlags = expected_flags;
362 res.quiet = quiet;

364 res.failed = false;
